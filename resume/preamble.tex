\makebibcategory{contributed}{Contributed publications}{Those publications on which I had a significant contribution to the work presented.}
\makebibcategory{lvc}{LIGO-Virgo publications}{Those papers published by the LIGO Scientific Collaboration and The Virgo Collaboration on which I have authorship credits.}
